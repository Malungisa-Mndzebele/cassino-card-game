# Systemd Service Template for Casino Card Game Backend
# File: /etc/systemd/system/casino-backend.service
# Note: Requires root access to install

[Unit]
Description=Casino Card Game Backend (FastAPI)
After=network.target mysql.service

[Service]
Type=simple
User=mawdqtvped
Group=mawdqtvped
WorkingDirectory=/home/mawdqtvped/khasinogaming.com/cassino
Environment="PATH=/home/mawdqtvped/khasinogaming.com/cassino/venv/bin"
ExecStart=/home/mawdqtvped/khasinogaming.com/cassino/venv/bin/python start_production.py
Restart=always
RestartSec=10

# Logging
StandardOutput=append:/home/mawdqtvped/khasinogaming.com/cassino/logs/stdout.log
StandardError=append:/home/mawdqtvped/khasinogaming.com/cassino/logs/stderr.log

[Install]
WantedBy=multi-user.target

# Installation instructions:
# 1. Create logs directory: mkdir -p /home/mawdqtvped/khasinogaming.com/cassino/logs
# 2. Copy this file to: /etc/systemd/system/casino-backend.service
# 3. Reload systemd: sudo systemctl daemon-reload
# 4. Enable service: sudo systemctl enable casino-backend
# 5. Start service: sudo systemctl start casino-backend
# 6. Check status: sudo systemctl status casino-backend
# 7. View logs: journalctl -u casino-backend -f
